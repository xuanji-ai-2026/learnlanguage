```bash
cat > backend/models/progress.py << 'EOF'
"""
进度追踪数据模型
"""
from pydantic import BaseModel, Field
from typing import Optional
from datetime import datetime


class ProgressBase(BaseModel):
    """进度基础模型"""
    user_id: str = Field(..., description="用户ID")
    vocabulary_id: str = Field(..., description="词汇ID")
    status: str = Field(..., description="学习状态（learning/mastered）")
    review_count: int = Field(default=0, description="复习次数")


class Progress(ProgressBase):
    """进度完整模型"""
    id: str = Field(default_factory=lambda: str(uuid.uuid4()), description="进度ID")
    last_reviewed_at: Optional[datetime] = Field(None, description="最后复习时间")
    next_review_at: Optional[datetime] = Field(None, description="下次复习时间")
    created_at: datetime = Field(default_factory=datetime.now, description="创建时间")
    updated_at: datetime = Field(default_factory=datetime.now, description="更新时间")

    class Config:
        from_attributes = True
EOF
```
